CREATE TABLE JOB (
		JOB_ID TEXT(2000000000) NOT NULL,
		JOB_NAME TEXT(2000000000) NOT NULL,
		COMPANYPROFILE_ID TEXT(2000000000) NOT NULL,
		SALARY_RATE INTEGER NOT NULL,
		POSITION_TYPE TEXT(2000000000) NOT NULL,
		LOCATION TEXT(2000000000) NOT NULL,
		JOB_DESCRIPTION TEXT(2000000000) NOT NULL,
		STATUS TEXT(2000000000) NOT NULL
	);

CREATE UNIQUE INDEX sqlite_autoindex_JOB_1 ON JOB (JOB_ID ASC);
