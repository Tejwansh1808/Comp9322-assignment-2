CREATE TABLE  USERPROFILE 
   (USERID CHAR(1000) PRIMARY KEY NOT NULL, 
	NAME CHAR(1000) NOT NULL, 
	EMAIL CHAR(1000) NOT NULL, 
	DOB CHAR(1000) NOT NULL , 
	CURRENT_POSITION CHAR(1000), 
	CURRENT_COMPANY CHAR(1000), 
	HIGHEST_EDUCATION CHAR(1000) NOT NULL , 
	PAST_EXPERIENCE CHAR(1000), 
	PROFESSIONAL_SKILLS CHAR(1000) NOT NULL , 
	RESUME CHAR(1000), 
	CV CHAR(1000)) ;
